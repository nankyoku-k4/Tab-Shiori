function setBookmark(emoji, bgColor, textColor) {
    // 1. タブのタイトルを変更
    document.title = `${emoji} ーーーーー しおり ーーーーー`;
    
    // 2. SVGで目立つファビコンを動的に生成して差し替え
    const faviconUrl = `data:image/svg+xml,<svg xmlns=%22http://www.w3.org/2000/svg%22 viewBox=%220 0 100 100%22><rect width=%22100%22 height=%22100%22 fill=%22%23${bgColor}%22/><text y=%2275%22 x=%2250%22 font-size=%2275%22 text-anchor=%22middle%22 fill=%22${textColor}%22 font-weight=%22bold%22>┃</text></svg>`;
    
    document.getElementById('favicon').setAttribute('href', faviconUrl);
}
